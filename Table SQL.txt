 CREATE TABLE "BOARD" 
   (	"SEQ" NUMBER NOT NULL ENABLE, 
	"ID" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"SUBJECT" VARCHAR2(255 BYTE) NOT NULL ENABLE, 
	"CONTENT" VARCHAR2(4000 BYTE) NOT NULL ENABLE, 
	"HIT" NUMBER DEFAULT 0, 
	"LOGTIME" DATE DEFAULT sysdate, 
	"FILENAME" VARCHAR2(20 BYTE), 
	"ORIGINAL_FILENAME" VARCHAR2(20 BYTE)
   )

 CREATE TABLE "BOARD_COMMENT" 
   (	"RNO" NUMBER, 
	"BNO" NUMBER NOT NULL ENABLE, 
	"COMMENT_WRITER" VARCHAR2(50 BYTE) NOT NULL ENABLE, 
	"COMMENT_CONTENT" VARCHAR2(1000 BYTE) NOT NULL ENABLE, 
	"COMMENT_REGDATE" DATE DEFAULT sysdate, 
	"COMMENT_DEPTH" NUMBER(*,0) DEFAULT '0', 
	"COMMENT_GROUP" NUMBER(*,0) DEFAULT '0', 
	"COMMENT_ORDER" NUMBER(*,0) DEFAULT '0'
   )

 CREATE TABLE "MEMBER" 
   (	"ID" VARCHAR2(30 BYTE) NOT NULL ENABLE, 
	"PW" VARCHAR2(64 BYTE) NOT NULL ENABLE, 
	"SALT" VARCHAR2(24 BYTE) NOT NULL ENABLE
   )
